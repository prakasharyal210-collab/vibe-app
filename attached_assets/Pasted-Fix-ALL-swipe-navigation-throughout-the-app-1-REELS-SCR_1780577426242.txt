5. IMPLEMENTATION:

For Reels (nested swipes):
Use react-native-gesture-handler with 
simultaneous gesture support:

import { 
  GestureDetector, 
  Gesture,
  GestureHandlerRootView 
} from 'react-native-gesture-handler'

// Vertical gesture for reel scrolling
const verticalGesture = Gesture.Pan()
  .onUpdate((e) => {
    if (Math.abs(e.translationY) > 
        Math.abs(e.translationX)) {
      handleReelScroll(e.translationY)
    }
  })
  .onEnd((e) => {
    if (e.translationY < -50) nextReel()
    if (e.translationY > 50) prevReel()
  })

// Horizontal gesture for tab switching
const horizontalGesture = Gesture.Pan()
  .onUpdate((e) => {
    if (Math.abs(e.translationX) > 
        Math.abs(e.translationY)) {
      handleTabSwipe(e.translationX)
    }
  })
  .onEnd((e) => {
    if (e.translationX < -50) switchToForYou()
    if (e.translationX > 50) switchToFollowing()
  })

// Race gesture - first one wins
const gesture = Gesture.Race(
  verticalGesture,
  horizontalGesture
)
