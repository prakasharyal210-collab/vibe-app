4. REAL TIME PROFILE UPDATES:
Subscribe to posts AND reels tables:

// New post by this user
supabase.channel('profile-posts')
  .on('postgres_changes', {
    event: 'INSERT',
    schema: 'public',
    table: 'posts',
    filter: `user_id=eq.${userId}`
  }, (payload) => {
    addPostToGrid(payload.new)
    incrementPostsCount()
  })
  .subscribe()

// New reel by this user  
supabase.channel('profile-reels')
  .on('postgres_changes', {
    event: 'INSERT',
    schema: 'public',
    table: 'reels',
    filter: `user_id=eq.${userId}`
  }, (payload) => {
    addReelToGrid(payload.new)
    incrementPostsCount()
  })
  .subscribe()
