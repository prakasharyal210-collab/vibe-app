MATCHES TAB - Use new Supabase functions:

1. LOAD MATCHES:
const loadMatches = async () => {
  setLoading(true)
  const { data, error } = await supabase
    .rpc('get_my_vibe_matches', {
      p_user_id: userId,
      p_limit: 50
    })
  if (data) setMatches(data)
  setLoading(false)
}

7. REAL TIME NEW MATCHES:
supabase
  .channel('new-matches')
  .on('postgres_changes', {
    event: 'UPDATE',
    schema: 'public',
    table: 'vibe_matches',
    filter: `receiver_id=eq.${userId}`
  }, async (payload) => {
    if (payload.new.status === 'accepted') {
      // Reload matches
      await loadMatches()
      // Show toast notification
      showToast({
        type: 'match',
        message: 'New Vibe Match! 💜',
        onPress: () => switchToMatchesTab()
      })
    }
  })
  .subscribe()
