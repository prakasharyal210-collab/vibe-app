  // NO __DEV__ guard here
  try {
    const { data } = await supabase
      .from('app_config')
      .select('*')
    
    const config = {}
    data?.forEach(r => config[r.key] = r.value)
    
    if (config.force_update === 'true') {
      setForceUpdate(true) // Shows with skip
    }
    
    if (config.maintenance_mode === 'true') {
      setMaintenance(true) // Shows with skip
    }
  } catch (err) {
    console.log('Config check failed:', err)
  }
}
