FEED ADS (like Facebook):
When loading feed insert ads between posts:

const insertAdsInFeed = (posts, ads) => {
  const result = []
  posts.forEach((post, index) => {
    result.push(post)
    // Insert ad every 4 posts
    if ((index + 1) % 4 === 0 && ads.length > 0) {
      result.push({
        ...ads[Math.floor(index/4) % ads.length],
        isAd: true
      })
    }
  })
  return result
}

const loadFeedAds = async () => {
  const { data } = await supabase
    .rpc('get_feed_ads', {
      p_user_id: userId,
      p_ad_type: 'feed_post',
      p_limit: 5
    })
  
  // If no real ads use house ads
  if (!data || data.length === 0) {
    return getHouseAds()
  }
  return data
}

// Track when ad is shown
const onAdVisible = async (adId) => {
  if (adId.startsWith('house-')) return
  await supabase.rpc('track_ad_impression', {
    p_ad_id: adId,
    p_user_id: userId,
    p_impression_type: 'view',
    p_watch_duration: 0
  })
}

// Track ad click
const onAdClick = async (adId, ctaUrl) => {
  if (!adId.startsWith('house-')) {
    await supabase.rpc('track_ad_click', {
      p_ad_id: adId,
      p_user_id: userId
    })
  }
  // Open URL
  if (ctaUrl.startsWith('vibe://')) {
    handleDeepLink(ctaUrl)
  } else {
    Linking.openURL(ctaUrl)
  }
}
