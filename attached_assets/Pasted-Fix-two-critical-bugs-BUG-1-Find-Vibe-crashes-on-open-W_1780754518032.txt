const uploadPhoto = async (uri) => {
  try {
    const filename = `${userId}/${Date.now()}.jpg`
    const response = await fetch(uri)
    const blob = await response.blob()
    
    const { data, error } = await supabase.storage
      .from('posts')
      .upload(filename, blob, {
        contentType: 'image/jpeg',
        upsert: false
      })
    
    if (error) throw error
    
    const { data: urlData } = supabase.storage
      .from('posts')
      .getPublicUrl(filename)
    
    return urlData.publicUrl
  } catch (err) {
    console.log('Upload error:', err)
    return null
  }
}

2. SAVE POST TO DATABASE:
const createPost = async (mediaUrl, caption) => {
  try {
    const { data, error } = await supabase
      .from('posts')
      .insert({
        user_id: userId,
        media_url: mediaUrl,
        media_type: 'image',
        caption: caption,
        likes_count: 0,
        comments_count: 0,
        views_count: 0,
        created_at: new Date().toISOString()
      })
      .select()
      .single()
    
    if (error) throw error
    
    // Update posts count on profile
    await supabase
      .from('profiles')
      .update({ 
        posts_count: supabase.rpc('increment', 
          { row_id: userId }) 
      })
      .eq('id', userId)
    
    return data
  } catch (err) {
    console.log('Create post error:', err)
    return null
  }
}

const getFeed = async () => {
  try {
    // First try algorithm feed
    const { data } = await supabase
      .rpc('get_for_you_feed', {
        p_user_id: userId,
        p_limit: 20
      })
    
    if (data && data.length > 0) {
      setFeed(data)
      return
    }
    
    // Fallback - show ALL posts if feed empty
    const { data: allPosts } = await supabase
      .from('posts')
      .select(`
        *,
        profiles!user_id (
          username,
          avatar_url,
          is_verified
        )
      `)
      .order('created_at', { ascending: false })
      .limit(20)
    
    setFeed(allPosts || [])
  } catch (err) {
    console.log('Feed error:', err)
    setFeed([])
  }
}
