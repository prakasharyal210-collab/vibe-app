import AsyncStorage from '@react-native-async-storage/async-storage'

const SETUP_INTERVAL_DAYS = 60 // 2 months

const shouldShowSetup = async (userId: string) => {
  try {
    // Check last setup date
    const lastSetup = await AsyncStorage.getItem(
      `vibe_setup_${userId}`
    )
    
    if (!lastSetup) {
      // Never done setup → show it
      return true
    }
    
    const lastDate = new Date(lastSetup)
    const now = new Date()
    const daysDiff = Math.floor(
      (now.getTime() - lastDate.getTime()) / 
      (1000 * 60 * 60 * 24)
    )
    
    // Show again if 60+ days passed
    return daysDiff >= SETUP_INTERVAL_DAYS
    
  } catch {
    return true
  }
}

const markSetupComplete = async (userId: string) => {
  // Save completion date
  await AsyncStorage.setItem(
    `vibe_setup_${userId}`,
    new Date().toISOString()
  )
  
  // Also save to Supabase
  await supabase
    .from('user_settings')
    .upsert({
      user_id: userId,
      vibe_setup_last_shown: new Date().toISOString()
    })
}
