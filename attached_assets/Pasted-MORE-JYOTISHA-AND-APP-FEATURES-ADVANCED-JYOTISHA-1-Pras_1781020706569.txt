sql-- Prashna questions
CREATE TABLE prashna_readings (

  question TEXT,
  chart_time TIMESTAMP,
  reading TEXT,
  created_at TIMESTAMP DEFAULT NOW()
);

-- Muhurta selections
CREATE TABLE muhurta_requests (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),

  activity_type TEXT,

  preferred_date_to DATE,
  auspicious_times JSONB,
  created_at TIMESTAMP DEFAULT NOW()
);

-- Numerology profiles
CREATE TABLE numerology_profiles (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  user_id UUID REFERENCES profiles(id),
  life_path_number INTEGER,
  name_number INTEGER,
  destiny_number INTEGER,
  lucky_numbers INTEGER[],
  created_at TIMESTAMP DEFAULT NOW()
);

-- Pilgrimage planner
CREATE TABLE pilgrimage_plans (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  user_id UUID REFERENCES profiles(id),
  destination TEXT,
  recommended_date DATE,
  planetary_reason TEXT,
  prayers_to_do TEXT,
  created_at TIMESTAMP DEFAULT NOW()
);

-- Astro journal
CREATE TABLE astro_journal (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  user_id UUID REFERENCES profiles(id),

  mood TEXT,
  created_at TIMESTAMP DEFAULT NOW()
