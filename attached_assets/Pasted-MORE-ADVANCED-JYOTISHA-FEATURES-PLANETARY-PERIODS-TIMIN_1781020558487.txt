sql-- Dasha calendar
CREATE TABLE dasha_periods (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  user_id UUID REFERENCES profiles(id),
  mahadasha TEXT,
  antardasha TEXT,
  pratyantar TEXT,
  start_date DATE,
  end_date DATE,
  created_at TIMESTAMP DEFAULT NOW()
);

-- Transit alerts
CREATE TABLE transit_alerts (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  user_id UUID REFERENCES profiles(id),
  planet TEXT,
  from_sign TEXT,
  to_sign TEXT,
  transit_date DATE,
  alert_sent BOOLEAN DEFAULT FALSE,
  created_at TIMESTAMP DEFAULT NOW()
);

-- Family charts
CREATE TABLE family_charts (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  owner_id UUID REFERENCES profiles(id),
  member_name TEXT,
  relationship TEXT,
  birth_date DATE,
  birth_time TIME,
  birth_place TEXT,
  rashi TEXT,
  lagna TEXT,
  nakshatra TEXT,
  chart_data JSONB,
  created_at TIMESTAMP DEFAULT NOW()
);

-- Mantra tracker
CREATE TABLE mantra_tracker (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  user_id UUID REFERENCES profiles(id),
  planet TEXT,
  mantra TEXT,
  count INTEGER DEFAULT 0,
  date DATE,
  created_at TIMESTAMP DEFAULT NOW()
);

-- Karma points
CREATE TABLE karma_points (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  user_id UUID REFERENCES profiles(id),
  activity TEXT,
  points INTEGER,
  created_at TIMESTAMP DEFAULT NOW()
);

-- Spiritual streak
CREATE TABLE spiritual_streaks (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  user_id UUID REFERENCES profiles(id),
  streak_type TEXT,
  current_streak INTEGER DEFAULT 0,
  longest_streak INTEGER DEFAULT 0,
  last_activity DATE,
  created_at TIMESTAMP DEFAULT NOW()
);

-- Jyotisha courses
CREATE TABLE jyotisha_course_progress (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),

  course_level TEXT,

  completed_at TIMESTAMP,
  created_at TIMESTAMP DEFAULT NOW()
