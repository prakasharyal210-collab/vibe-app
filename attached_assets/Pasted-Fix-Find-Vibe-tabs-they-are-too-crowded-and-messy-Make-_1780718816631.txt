CODE:
<ScrollView
  horizontal
  showsHorizontalScrollIndicator={false}
  contentContainerStyle={{
    paddingHorizontal: 16,
    gap: 8,
    alignItems: 'center'
  }}
>
  {TABS.map((tab, index) => (
    <TouchableOpacity
      key={tab.key}
      onPress={() => switchToTab(index)}
      style={{
        paddingHorizontal: 16,
        paddingVertical: 8,
        borderRadius: 20,
        minWidth: 70,
        alignItems: 'center',
        backgroundColor: activeTab === index 
          ? gradientColor 
          : 'transparent',
        borderWidth: activeTab === index ? 0 : 1,
        borderColor: 'rgba(255,255,255,0.1)'
      }}
    >
      <Text style={{ fontSize: 18 }}>
        {tab.emoji}
      </Text>
      <Text style={{
        fontSize: 11,
        fontWeight: '600',
        color: activeTab === index 
          ? '#fff' 
          : '#6B7280',
        marginTop: 2
      }}>
        {tab.label}
      </Text>
    </TouchableOpacity>
  ))}
</ScrollView>
