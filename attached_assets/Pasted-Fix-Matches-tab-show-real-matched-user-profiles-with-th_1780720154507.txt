1. Load real matches from Supabase:
const loadMatches = async () => {
  const { data } = await supabase
    .rpc('get_my_vibe_matches', {
      p_user_id: userId,
      p_limit: 50
    })
  
  // For each match load their full profile
  const matchesWithProfiles = await Promise.all(
    data.map(async (match) => {
      const { data: profile } = await supabase
        .rpc('get_full_profile', {
          p_username: match.username,
          p_viewer_id: userId
        })
      return {
        ...match,
        profile: profile?.[0]
      }
    })
  )
  setMatches(matchesWithProfiles)
}

8. REAL TIME UPDATES:
Subscribe to vibe_matches table:
supabase.channel('my-matches')
  .on('postgres_changes', {
    event: 'UPDATE',
    schema: 'public',
    table: 'vibe_matches',
    filter: `receiver_id=eq.${userId}`
  }, (payload) => {
    if (payload.new.status === 'accepted') {
      // New match! Add to list
      addNewMatch(payload.new)
      // Show match notification
      showMatchToast(payload.new)
    }
  })
  .subscribe()
