Add "Couple of the Month" competition feature inside the existing Couple tab in Gundruk app.

CONTEXT:
- All DB access through artifacts/api-server (never direct Supabase from mobile)
- Couple tab already exists at artifacts/mobile/app/couple/index.tsx
- Dark theme, purple accents (#8B5CF6)

STEP 1 — DATABASE (run in Supabase):

CREATE TABLE couple_competitions (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  month INT NOT NULL,
  year INT NOT NULL,
  couple_id UUID REFERENCES couple_links(id) ON DELETE CASCADE,
  couple_name TEXT NOT NULL,
  cover_photo_url TEXT,
  vote_count INT DEFAULT 0,
  entered_at TIMESTAMPTZ DEFAULT NOW(),
  UNIQUE(couple_id, month, year)
);

CREATE TABLE couple_competition_votes (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  voter_id UUID REFERENCES users(id) ON DELETE CASCADE,
  competition_id UUID REFERENCES couple_competitions(id) ON DELETE CASCADE,
  voted_at TIMESTAMPTZ DEFAULT NOW(),
  UNIQUE(voter_id, competition_id)
);

CREATE TABLE couple_competition_winners (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  couple_id UUID REFERENCES couple_links(id) ON DELETE CASCADE,
  month INT NOT NULL,
  year INT NOT NULL,
  rank INT NOT NULL CHECK (rank IN (1, 2, 3)),
  vote_count INT NOT NULL,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

ALTER TABLE couple_competitions DISABLE ROW LEVEL SECURITY;
ALTER TABLE couple_competition_votes DISABLE ROW LEVEL SECURITY;
ALTER TABLE couple_competition_winners DISABLE ROW LEVEL SECURITY;

STEP 2 — API ROUTES (in artifacts/api-server/src/routes/couple.ts, add to existing file):

- POST /api/couple/competition/enter — enter this month's competition (body: coupleName, coverPhotoUrl)
- GET /api/couple/competition/leaderboard — get top 10 couples for current month with vote counts, days together, partner info
- POST /api/couple/competition/vote/:competitionId — vote for a couple (one vote per user per month)
- DELETE /api/couple/competition/vote/:competitionId — unvote
- GET /api/couple/competition/winners — get all time hall of fame winners
- GET /api/couple/competition/my-entry — get current couple's entry for this month
