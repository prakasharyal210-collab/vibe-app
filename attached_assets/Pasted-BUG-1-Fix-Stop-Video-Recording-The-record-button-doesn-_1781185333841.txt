// Simple boomerang using expo-video-thumbnails
import * as VideoThumbnails from 'expo-video-thumbnails';

const createBoomerang = async (videoUri) => {
  // Extract frames from video
  const frames = [];
  const duration = 2000; // 2 seconds in ms
  
  for (let time = 0; time <= duration; time += 100) {
    const { uri } = await VideoThumbnails.getThumbnailAsync(videoUri, {
      time: time,
      quality: 0.8,
    });
    frames.push(uri);
  }
  
  // Reverse frames for backward play
  const reversedFrames = [...frames].reverse();
  
  // Combine: forward + backward frames
  const boomerangFrames = [...frames, ...reversedFrames];
  
  // Convert frames back to video
  // Return original video URI as fallback if frame manipulation not available
  return videoUri;
};

BUG 3 — Add Zoom In/Zoom Out:
Add zoom controls to camera:
javascriptconst [zoom, setZoom] = useState(0); // 0 = no zoom, 1 = max zoom

// Pinch to zoom gesture
import { PinchGestureHandler } from 'react-native-gesture-handler';

const onPinchGestureEvent = (event) => {
  const scale = event.nativeEvent.scale;
  const newZoom = Math.min(Math.max(zoom + (scale - 1) * 0.1, 0), 1);
  setZoom(newZoom);
};

<PinchGestureHandler onGestureEvent={onPinchGestureEvent}>
  <Camera
    ref={cameraRef}
    zoom={zoom} // expo-camera zoom prop
    style={styles.camera}
  />
</PinchGestureHandler>

// Zoom slider on right side of camera
<View style={styles.zoomSlider}>
  <Text>🔍+</Text>
  <Slider
    style={styles.verticalSlider}
    value={zoom}
    minimumValue={0}
    maximumValue={1}
    onValueChange={setZoom}
    vertical={true}
  />
  <Text>🔍-</Text>
</View>

// Zoom level indicator
<Text style={styles.zoomLabel}>
  {zoom === 0 ? '1x' : zoom < 0.3 ? '2x' : zoom < 0.6 ? '3x' : '5x'}
</Text>

// Tap zoom level to cycle through: 1x → 2x → 3x → 0.5x
<TouchableOpacity onPress={cycleZoom}>
  <Text>{zoomLabel}x</Text>
</TouchableOpacity>

const cycleZoom = () => {
  const levels = [0, 0.2, 0.5, 0.8];
  const currentIndex = levels.indexOf(zoom);
  const nextIndex = (currentIndex + 1) % levels.length;
  setZoom(levels[nextIndex]);
};
