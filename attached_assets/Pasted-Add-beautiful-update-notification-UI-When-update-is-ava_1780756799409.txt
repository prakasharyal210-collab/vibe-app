IMPLEMENTATION in app/_layout.tsx:

const [updateAvailable, setUpdateAvailable] = 
  useState(false)
const [updateDownloaded, setUpdateDownloaded] = 
  useState(false)
const [downloading, setDownloading] = 
  useState(false)
const [downloadProgress, setDownloadProgress] = 
  useState(0)
const [forceUpdate, setForceUpdate] = 
  useState(false)
const [whatsNew, setWhatsNew] = useState([])

useEffect(() => {
  checkForUpdates()
  checkServerVersion()
}, [])

const checkForUpdates = async () => {
  try {
    if (__DEV__) return
    
    const update = await 
      Updates.checkForUpdateAsync()
    
    if (update.isAvailable) {
      setUpdateAvailable(true)
      // Auto download in background
      setDownloading(true)
      await Updates.fetchUpdateAsync()
      setUpdateDownloaded(true)
      setDownloading(false)
    }
  } catch (err) {
    console.log('Update check failed:', err)
  }
}

const checkServerVersion = async () => {
  try {
    const { data } = await supabase
      .from('app_config')
      .select('*')
    
    const config = {}
    data?.forEach(r => config[r.key] = r.value)
    
    // Check force update
    if (config.force_update === 'true') {
      setForceUpdate(true)
    }
    
    // Get whats new
    if (config.whats_new) {
      setWhatsNew(JSON.parse(config.whats_new))
    }
    
    // Check maintenance
    if (config.maintenance_mode === 'true') {
      showMaintenanceScreen(
        config.maintenance_message
      )
    }
  } catch (err) {
    console.log('Version check failed:', err)
  }
}

const handleUpdate = async () => {
  try {
    if (updateDownloaded) {
      // Already downloaded - just reload
      await Updates.reloadAsync()
    } else {
      // Download then reload
      setDownloading(true)
      await Updates.fetchUpdateAsync()
      await Updates.reloadAsync()
    }
  } catch (err) {
    // If OTA fails open Play Store
    Linking.openURL(
      'market://details?id=com.vibeapp.vibe'
    )
  }
}
