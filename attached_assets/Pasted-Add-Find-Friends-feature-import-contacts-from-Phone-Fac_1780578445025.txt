1. PHONE CONTACTS:

Ask permission to access contacts:
import * as Contacts from 'expo-contacts'

const { status } = await Contacts.requestPermissionsAsync()

if (status === 'granted') {
  const { data } = await Contacts.getContactsAsync({
    fields: [
      Contacts.Fields.PhoneNumbers,
      Contacts.Fields.Emails,
      Contacts.Fields.Name,
      Contacts.Fields.Image
    ]
  })
  
  // Extract phone numbers and emails
  const phoneNumbers = data
    .flatMap(c => c.phoneNumbers || [])
    .map(p => p.number)
    
  const emails = data
    .flatMap(c => c.emails || [])
    .map(e => e.email)
  
  // Find Vibe users with these numbers/emails
  findVibeUsersFromContacts(phoneNumbers, emails)
}

2. FACEBOOK FRIENDS:

Use Facebook Login SDK:
import * as Facebook from 'expo-facebook'

const connectFacebook = async () => {
  await Facebook.initializeAsync({
    appId: 'YOUR_FB_APP_ID'
  })
  
  const { type, token } = 
    await Facebook.logInWithReadPermissionsAsync({
      permissions: ['public_profile', 'user_friends']
    })
    
  if (type === 'success') {
    // Get Facebook friends
    const response = await fetch(
      `https://graph.facebook.com/me/friends?
       access_token=${token}`
    )
    const { data: fbFriends } = await response.json()
    
    // Get user's own FB profile
    const profileRes = await fetch(
      `https://graph.facebook.com/me?
       fields=id,name,email,picture&
       access_token=${token}`
    )
    const fbProfile = await profileRes.json()
    
    // Save FB connection to Supabase
    await supabase.from('social_connections').upsert({
      user_id: myUserId,
      platform: 'facebook',
      platform_id: fbProfile.id,
      platform_username: fbProfile.name,
      access_token: token
    })
    
    // Find FB friends on Vibe
    findVibeUsersFromFacebook(fbFriends)
  }
}

4. SAVE SOCIAL CONNECTIONS TO SUPABASE:
Already have social_connections table or create:

create table if not exists social_connections (
  id uuid default gen_random_uuid() primary key,
  user_id uuid references profiles(id) on delete cascade,
  platform text not null, -- facebook, tiktok, instagram
  platform_id text,
  platform_username text,
  platform_email text,
  access_token text,
  connected_at timestamp default now(),
  unique(user_id, platform)
);
